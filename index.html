                        example: true

                        type: boolean
                        description: Flags if bulk delete of nodes is available.
                        example: true
                    roleMapping:
                        type: boolean
                        description: Flags if role mapping support is available.
                        example: true
                    bulkDelete:
                        type: boolean
                        description: Flags if bulk delete feature is available.
                        example: true
